# EKIO Complete Six-Theme System for R/ggplot2 library(ggplot2) library(scales) library(extrafont) # ==== EKIO MODERN PREMIUM (PRIMARY THEME) ==== ekio_modern_premium <- list( colors = c("#f4f7fb", "#e3ecf5", "#d2e1ef", "#a0bfdb", "#6d9dc7", "#2C6BB3", "#265c9a", "#204d81", "#1a3e68"), primary = "#2C6BB3", font = "Avenir", categorical = c("#2C6BB3", "#1abc9c", "#f39c12", "#e74c3c", "#9b59b6") ) # ==== UNIVERSAL THEME FUNCTION ==== theme_ekio <- function(style = "modern_premium", base_size = 12) { # Theme configurations themes <- list( "modern_premium" = list(primary = "#2C6BB3", font = "Avenir", colors = c("#f4f7fb", "#e3ecf5", "#2C6BB3")), "academic_authority" = list(primary = "#0F3A65", font = "Lato", colors = c("#f2f4f7", "#e0e6ed", "#0F3A65")), "sophisticated_unique" = list(primary = "#5F9EA0", font = "Avenir", colors = c("#f6fafa", "#e8f4f4", "#5F9EA0")), "institutional_oxford" = list(primary = "#1E5F9F", font = "Helvetica Neue", colors = c("#f3f6fa", "#e2eaf4", "#1E5F9F")), "professional_deep" = list(primary = "#0F3A65", font = "Helvetica Neue", colors = c("#f2f4f7", "#e0e6ed", "#0F3A65")), "premium_steel" = list(primary = "#4682B4", font = "Avenir", colors = c("#f4f7fa", "#e1eaf2", "#4682B4")) ) theme_config <- themes[[style]] %||% themes$modern_premium theme_minimal(base_size = base_size, base_family = theme_config$font) + theme( text = element_text(family = theme_config$font, color = "#2c3e50"), plot.title = element_text(size = base_size * 1.6, color = theme_config$primary, margin = margin(b = 20)), plot.subtitle = element_text(size = base_size * 1.1, color = "#7f8c8d", margin = margin(b = 25)), plot.caption = element_text(size = base_size * 0.8, color = "#7f8c8d", hjust = 0, margin = margin(t = 15)), axis.title = element_text(size = base_size * 0.9, color = "#7f8c8d"), axis.text = element_text(size = base_size * 0.8, color = "#2c3e50"), panel.grid.major = element_line(color = theme_config$colors[2], size = 0.5), panel.grid.minor = element_blank(), panel.background = element_rect(fill = "white", color = NA), legend.position = "bottom", plot.margin = margin(25, 25, 25, 25) ) } # ==== SEQUENTIAL COLOR SCALES ==== scale_fill_ekio <- function(style = "modern_premium", n = 7, ...) { colors <- switch(style, "modern_premium" = c("#f4f7fb", "#d2e1ef", "#6d9dc7", "#2C6BB3", "#204d81"), "academic_authority" = c("#f2f4f7", "#ced8e3", "#7090b5", "#0F3A65", "#092238"), "sophisticated_unique" = c("#f6fafa", "#d9efef", "#96ced1", "#5F9EA0", "#2c4e50"), c("#f4f7fb", "#d2e1ef", "#6d9dc7", "#2C6BB3", "#204d81") # default ) scale_fill_gradientn(colors = colorRampPalette(colors)(n), ...) } # ==== USAGE EXAMPLES ==== # Standard client presentation client_plot <- ggplot(mtcars, aes(x = wt, y = mpg)) + geom_point(color = "#2C6BB3", size = 3, alpha = 0.8) + theme_ekio("modern_premium") + labs( title = "Brazilian Urban Vehicle Usage Analysis", subtitle = "Fuel efficiency patterns across metropolitan areas", x = "Vehicle Weight (tons)", y = "Fuel Efficiency (km/L)", caption = "Source: DETRAN Municipal Data | EKIO Analytics" ) # Brazilian state choropleth with EKIO styling brazil_map <- states %>% left_join(economic_data, by = "code_state") %>% ggplot() + geom_sf(aes(fill = gdp_index), color = "white", size = 0.3) + scale_fill_ekio("modern_premium", 7) + theme_ekio("modern_premium") + theme( axis.text = element_blank(), axis.ticks = element_blank(), panel.grid = element_blank() ) + labs( title = "Brazilian Regional Economic Performance", subtitle = "GDP Index by state with EKIO sequential scale", fill = "GDP Index", caption = "Source: IBGE Regional Accounts | EKIO Analytics" )
